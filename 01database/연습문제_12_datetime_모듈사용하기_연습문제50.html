문제 30

오늘 날짜에서 '월 기준 주차'를 계산하는 `week_of_month` 함수를 작성하고 출력하세요.

from datetime import date

def week_of_month(d: date, week_start: int = 0) -> int:
    first = d.replace(day=1)
    offset = (first.weekday() - week_start) % 7
    return (d.day + offset - 1) // 7 + 1

    interest = principal * annual_rate * (days / base_days)

from datetime import datetime, date, time

]

print('예상 이자(교육용):', round(interest))
